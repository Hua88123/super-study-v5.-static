-- 超能游学报价系统 V8 Supabase 建表 SQL
-- Supabase → SQL Editor → New query → 粘贴全部运行

create table if not exists public.super_study_data (
  id text primary key,
  payload jsonb not null,
  updated_at timestamptz default now()
);

create table if not exists public.super_study_employees (
  id text primary key,
  name text not null,
  login_code text unique not null,
  is_active boolean default true,
  bound_device_id text,
  bound_device_name text,
  created_at timestamptz default now(),
  last_login_at timestamptz
);

-- V8 通过 Vercel API 访问 Supabase，密钥只放在 Vercel 环境变量。
alter table public.super_study_data disable row level security;
alter table public.super_study_employees disable row level security;
